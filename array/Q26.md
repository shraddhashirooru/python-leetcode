class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        i=1
        while i<len(nums):
            if nums[i-1]==nums[i]:
                del nums[i]
            else:
                i=i+1
        return len(nums)

class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        a=set()
        i=0
        while i<len(nums):
            if nums[i] not in a:
                a.add(nums[i])
                i=i+1
            else:
                del nums[i]
        return len(nums)

class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        fast = slow = count = 0
        while fast<len(nums):
            if nums[fast]==nums[slow]:
                fast+=1
            else:
                count+=1
                slow+=1
                nums[slow]=nums[fast]
                fast+=1
        return count+1
